\documentclass{./resume}

% ---
% Command Definitions
% ---

% Define to switch between generating a private confidential document 
% or a public document to put in the web.
\newcommand{\PRIVATERESUME}

% Display private info
\newcommand{\privateinfo}
{
	% Only add address and phone for private documents.
	\resumesubtitle{my private address}{first.last@email.com}
	
	% Only add address and phone for private documents.
	\resumesubtitle{City, State, Country}{123-456-7890}
	
	% Put the online urls for all documents.
	\resumesubtitle{linkedin.com/in/firstlast}{firstlast.blog.com} 
}

% Display public info
\newcommand{\publicinfo}
{
	% Only add city and e-mail address for public documents.
	\resumesubtitle{City, State, Country}{first.last@email.com}  
	
	% Put the online urls for all documents.
	\resumesubtitle{linkedin.com/in/firstlast}{firstlast.blog.com} 
}
